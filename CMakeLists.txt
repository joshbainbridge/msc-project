CMAKE_MINIMUM_REQUIRED( VERSION 2.8 )
PROJECT( "msc-project" )

set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} ${CMAKE_SOURCE_DIR}/cmake $ENV{CMAKE_MODULE_PATH})

SET( CXX )
SET( FLAGS )

set( SRC src)
set( INC include)

SET( PROJ_SOURCES
  ${SRC}/main.cpp
  )
SET( PROJ_HEADERS
  )

ADD_EXECUTABLE( ${CMAKE_PROJECT_NAME} ${PROJ_SOURCES} ${PROJ_HEADERS} )

FIND_PACKAGE( Boost COMPONENTS system filesystem iostreams chrono python REQUIRED )
FIND_PACKAGE( TBB REQUIRED )
FIND_PACKAGE( Embree REQUIRED )
FIND_PACKAGE( OpenGL REQUIRED )
FIND_PACKAGE( GLFW REQUIRED )
FIND_PACKAGE( OpenEXR REQUIRED )
FIND_PACKAGE( OpenColorIO REQUIRED )
FIND_PACKAGE( OIIO REQUIRED )
FIND_PACKAGE( PythonLibs REQUIRED )

IF( ${CMAKE_SYSTEM_NAME} MATCHES "Windows" OR "Linux" )
  
  FIND_PACKAGE( GLEW REQUIRED )

  INCLUDE_DIRECTORIES( ${GLEW_INCLUDE_DIR} )
  TARGET_LINK_LIBRARIES( ${CMAKE_PROJECT_NAME} ${GLEW_LIBRARY} )

ENDIF()

SET(CMAKE_CXX_COMPILER ${CXX})
SET(CMAKE_CXX_FLAGS ${FLAGS})

INCLUDE_DIRECTORIES(
  include
  ${Boost_INCLUDE_DIRS}
  ${TBB_INCLUDE_DIRS}
  ${EMBREE_INCLUDE_DIRS}
  ${GLFW_INCLUDE_DIR}
  ${OPENEXR_INCLUDE_PATHS}
  ${OCIO_INCLUDES}
  ${OIIO_INCLUDES}
  ${PYTHON_INCLUDE_DIRS}
  )

TARGET_LINK_LIBRARIES( ${CMAKE_PROJECT_NAME}
  ${Boost_LIBRARIES}
  ${TBB_LIBRARIES}
  ${EMBREE_LIBRARIES}
  ${GLFW_LIBRARIES}
  ${OPENGL_LIBRARIES}
  ${OPENEXR_LIBRARIES}
  ${OCIO_LIBRARIES}
  ${OIIO_LIBRARIES}
  ${PYTHON_LIBRARIES}
  )
